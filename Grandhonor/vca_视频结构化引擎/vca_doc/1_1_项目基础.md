  + libsdl2-2.0-0 
  + liblapacke 
  + libatlas-base-dev 
  + libuuid1
  + zlib1g 
  + librdkafka1 
  + libeigen3-dev
  + nvidia
    + cuda_11.4.4_470.82.01_linux
    + cudnn-11.4-linux-x64-v8.2.4.15
    + TensorRT-8.2.5.1.Linux.x86_64-gnu.cuda-11.4.cudnn8.2
